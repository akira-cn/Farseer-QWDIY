Ajax静态用法 get

/**
 * 静态get方法
 * @method get
 * @static
 * @param {String|Form} url 请求的地址
 * @param {String|Json|Form} data (Optional)发送的数据
 * @param {Function} callback 请求完成后的回调
 * @returns {Ajax}
 * @example
	QW.Ajax.get('http://localhost/demos/io/data/?type=json',{key: 'value'},function(e){alert(this.requester.responseText);});
 */
